function start()
	print("ECS lua start...")
	print("ecs injected object", hero_prefab.name)
	print("ecs injected object", monster_prefab.name)

	--组建Data
	local moveData = require("Logic/MoveData")
  	ECS.TypeManager.RegisterType("MoveData", moveData)
	ECS.TypeManager.RegisterType("DataA", {})
	ECS.TypeManager.RegisterType("DataB", {value = 7})
	moveData.transform = self.transform
	--moveData.hero_prefab = CS.UnityEngine.GameObject.Instantiate(hero_prefab,self.transform)
	--moveData.monster_prefab = CS.UnityEngine.GameObject.Instantiate(monster_prefab,self.transform)
	moveData.hero_prefab = hero_prefab
	moveData.monster_prefab = monster_prefab
	--数据装载到ArcheType
	local archetype = ECS.World.Active.entityManager:CreateArchetype({"MoveData","DataA"})
	local entities = ECS.World.Active.entityManager:CreateEntitiesByArcheType(archetype,32)
	--


	ECS.World.Active.entityManager:AddComponent(entities[20], "DataB")
	----创建 System
	ECS.World.Active:CreateSystem("MoveSystem")



end
